#include <stdio.h>
int main()
{
    char a = 3;
    //  3---0000000000000000000000000000011
    //3是一个整型，有4个字节，而char中只能存放1个字节，所以在存放的时候会发生截断
    //截断存放最低为，在这里存放00000011
    char b = 127;
    //127---0000000000000000000000001111111
    //b---01111111
    char c = a + b;
    //a+b
    //00000011
    //01111111
    //整型提升
    //a---0000000000000000000000000000011
    //b---0000000000000000000000001111111
    //c---0000000000000000000000010000010
    //截断
    //c---10000010
    //因为要输出整型类型的值，所以这里又要进行整型提升
    //c---1111111111111111111111110000010----补码
    //c---1111111111111111111111110000001----反码
    //c---1000000000000000000000001111110----原码
    //c=-126
    printf("%d\n",c);
    return 0;
}
