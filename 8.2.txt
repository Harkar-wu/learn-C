#include <stdio.h>
#include <string.h>

int main()
{
    char a[1000];
    int i;
    for (i = 0; i < 1000; i++)
    {
        a[i] = -1 - i;
    }
    printf("%d\n",strlen(a));
    return 0;
}
//char为有符号时取值范围为127至-128
//-1,-2,-3,-4............................-127,-128,127,126,125..........................,3,2,1,0
//0对应的字符为‘\0’
//所以strlen(a)=128+127=255
